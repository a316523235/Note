function quickSort(&$arr) {
    qSort($arr, 0, count($arr) - 1);
}

function qSort(&$arr, $i , $j) {
    $start = $i;
    $end = $j;

    if($i >= $j) {
        return;
    }

    $p = $arr[$i];

    while ($j > $i) {
        while ($j > $i && $arr[$j] >= $p) {
            $j--;
        }
        swap($arr, $i, $j);
        while($j > $i && $arr[$i] <= $p) {
            $i++;
        }
        swap($arr, $i, $j);
    }

    qSort($arr, $start, $i-1);
    qSort($arr, $i+1, $end);
}

function swap(&$arr, $i, $j) {
    $temp = $arr[$i];
    $arr[$i] = $arr[$j];
    $arr[$j] = $temp;
}
