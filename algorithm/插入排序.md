<?php

/**
 * 插入排序
 * @param array $arr
 */
function insertSort(&$arr) {
    for($i = 0; $i < count($arr); $i++) {
        $n = $arr[$i];
        for ($j = $i - 1; $j >= 0; $j--) {
            if($arr[$j] > $n) {
                $arr[$j+1] = $arr[$j];
            } else {
                break;
            }
        }
        $arr[$j+1] = $n;
    }
}


$arr = [5, 4, 6, -3, 7, 1];
insertSort($arr);
print_r($arr);
